    Simulate typing on a field, and when finished call the passed callback.

        simulateTyping: (field, value, callback) ->
        field.val ""
        arr = value.split ""
        i = 0
        delay = 180

        while i < arr.length
            setTimeout (() -> field.val(field.val() + "" + arr.shift())), delay * (i + 1)
            i++

        setTimeout callback, delay * (arr.length + 1)
